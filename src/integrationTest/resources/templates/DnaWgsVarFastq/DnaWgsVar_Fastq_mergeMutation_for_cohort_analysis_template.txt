# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

logFile=build/resources/integrationTest/output/log_files/DnaWgsVar_Fastq_gatkHaplotypeCaller_for_GA5_analysis.log
str=""
while [[('[')]][ $str = "" ]]
do
if [[('[')]][ -f $logFile  ]];
then
	str=$(grep -E "((Error Step: (Novoalign alignment|Index bam|Merge DNA bams|Mark duplicates|Index mkdup bam|Remove duplicates|Index rmdup bam|DNA QC metrics|Merge DNA QC|ABRA realignment|GATK haplotypecaller detection|SnpEff annotation|Remove temporary directories|Run gatkHaplotypeCaller|QC summary analysis))|(Successful Step: SnpEff annotation))" $logFile;)
fi
	echo Waiting for step: SnpEff annotation
	sleep 60
done

if [[('[')]][ $str == "*Error Step: SnpEff annotation*" ]];
then
	echo $(date) Error DNA mutation results from GA5:
	echo $str
else
	echo $(date) Confirm DNA mutation results from GA5
fi

echo `date` Begin Step: Merge mutation annotation...
/ngs/data/app/R/v3.5.0/bin/Rscript [(${jarPath})]/src/R/dna_rna_mutation_data_analysis.R -i build/resources/integrationTest/fastqSingle.tsv -d build/resources/integrationTest/output -t seqpurge+novoalign+abra_realign+picard+qc+gatkHaplotypeCaller
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Merge mutation annotation.
	sleep 8
else
	echo `date` Error Step: Merge mutation annotation.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Finish the job execution!
