# --- SGE options --- #

#$ -wd build/resources/integrationTest/output
#$ -N RnaCaptureVar_Fastq_gatkHaplotypeCaller_for_GA5_analysis

# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: GATK haplotypecaller detection...
/usr/lib/jvm/java-8-openjdk-amd64/bin/java -Xmx10g -Djava.io.tmpdir=build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/tmp -jar /usr/bin/gatk -T HaplotypeCaller -R /ngs/data/reference_genome/hg19/hg19_decoy/hg19.decoy.fa --intervals /ngs/data/S03723314_Padded.bed --input_file build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.realign.realign.recal.bam -dontUseSoftClippedBases -stand_call_conf 20.0 --out build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/GA5.gatkHaplotypeCaller.raw.vcf --validation_strictness SILENT
if [ $? -eq 0 ]
then
	echo `date` Successful Step: GATK haplotypecaller detection.
	sleep 8
else
	echo `date` Error Step: GATK haplotypecaller detection.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Begin Step: GATK haplotypecaller filtration...
/usr/lib/jvm/java-8-openjdk-amd64/bin/java -Xmx10g -Djava.io.tmpdir=build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/tmp -jar /usr/bin/gatk -T VariantFiltration -R /ngs/data/reference_genome/hg19/hg19_decoy/hg19.decoy.fa -V build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/GA5.gatkHaplotypeCaller.raw.vcf -window 35 -cluster 3 -filterName FS -filter "FS>30.0" -filterName QD -filter "QD<2.0" --out build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/GA5.gatkHaplotypeCaller.variants.vcf --validation_strictness SILENT
if [ $? -eq 0 ]
then
	echo `date` Successful Step: GATK haplotypecaller filtration.
	sleep 8
else
	echo `date` Error Step: GATK haplotypecaller filtration.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Begin Step: SnpEff annotation...
/usr/bin/python [(${jarPath})]/src/python/vcf_snpeff_annotation.py -s GA5 -i build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/GA5.gatkHaplotypeCaller.variants.vcf -o build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/GA5.gatkHaplotypeCaller.variants.pass.annotation.tsv -t gatkHaplotypeCaller --canonical /ngs/data/reference_genome/GRCh37/Annotation/prefer_ensembl_transcript.txt --transvar /usr/bin/transvar -r hg19 --snpsift /usr/bin/snpsift --snpsift_db /ngs/data/SnpEff/snpEff_v4.3p/snpEff/db --dbnsfp --dbsnp --coding_only --pass_only
if [ $? -eq 0 ]
then
	echo `date` Successful Step: SnpEff annotation.
	sleep 8
else
	echo `date` Error Step: SnpEff annotation.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: Remove temporary directories...
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.realign.realign.recal.table
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.insertsize.metrics
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.align.metrics
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.Aligned.out.bam.bai
rm -rf build/resources/integrationTest/output/GA5/gatkHaplotypeCaller/tmp
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.mkdup.bam.bai
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.realign.realign.bam.bai
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.mkdup.bam
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.bam.bai
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.rmdup.pileup
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.rmdup.hs.metrics
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.metrics
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.realign.realign_interval.list
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.bam
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.bam.bai
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.quality.metrics
rm -rf build/resources/integrationTest/output/GA5/tmp
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.rmdup.splitRead.realign.realign.bam
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.gcbias.metrics
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.gc.summary.metrics
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.star.sorted.bam
rm -rf build/resources/integrationTest/output/GA5/bam/GA5.Aligned.out.bam
rm -rf build/resources/integrationTest/output/GA5/qc/GA5.star.sorted.mkdup.hs.metrics
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Finish the job execution!
