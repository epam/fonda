#$ -q main.q
#$ -R y
#$ -pe threaded 4
#$ -m a

# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: Mutect2 detection...
/usr/bin/gatk Mutect2 -R /ngs/data/reference_genome/hg19/hg19_decoy/hg19.decoy.fa -I /ngs/data/demo/test/fastq_data/GA5_0001_L002_R1_003.bam -I /ngs/data/demo/test/fastq_data/GA5_0001_L002_R1_003.bam -normal GA51 -L /ngs/data/test_data_Padded.bed -pon /ngs/data/db/mormal_panel --bam-output build/resources/integrationTest/output/GA52/mutect2/GA52.mutect2.bamout.bam --f1r2-tar-gz build/resources/integrationTest/output/GA52/mutect2/GA52.mutect2.f1r2.tar.gz -O build/resources/integrationTest/output/GA52/mutect2/GA52.mutect2.somatic.variants.vcf
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Mutect2 detection.
	sleep 8
else
	echo `date` Error Step: Mutect2 detection.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: SnpEff annotation...
/usr/bin/python [(${jarPath})]/src/python/vcf_snpeff_annotation.py -s GA52 -i build/resources/integrationTest/output/GA52/mutect2/GA52.mutect2.somatic.variants.vcf -o build/resources/integrationTest/output/GA52/mutect2/GA52.mutect2.somatic.variants.pass.annotation.tsv -t mutect2 --canonical /ngs/data/reference_genome/GRCh37/Annotation/prefer_ensembl_transcript.txt --transvar /usr/bin/transvar -r hg19 --snpsift /usr/bin/snpsift --snpsift_db /ngs/data/SnpEff/snpEff_v4.3p/snpEff/db --dbnsfp --dbsnp --coding_only --pass_only
if [ $? -eq 0 ]
then
	echo `date` Successful Step: SnpEff annotation.
	sleep 8
else
	echo `date` Error Step: SnpEff annotation.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: Remove temporary directories...

if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Finish the job execution!
