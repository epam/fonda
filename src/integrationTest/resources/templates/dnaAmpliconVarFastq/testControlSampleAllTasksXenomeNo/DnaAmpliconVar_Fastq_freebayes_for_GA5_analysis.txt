#$ -N DnaAmpliconVar_Fastq_freebayes_for_GA5_analysis

#$ -m a

# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: Freebayes detection...
/usr/bin/freebayes -f /ngs/data/reference_genome/hg19/hg19_decoy/hg19.decoy.fa -F 0.05 -C 3 -m 20 -t /ngs/data/S03723314_Padded.bed build/resources/integrationTest/output/GA5/bam/GA5.merged.sorted.realign.realign.recal.bam > build/resources/integrationTest/output/GA5/freebayes/GA5.freebayes.variants.vcf
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Freebayes detection.
	sleep 8
else
	echo `date` Error Step: Freebayes detection.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Begin Step: SnpEff annotation...
/usr/bin/python [(${jarPath})]/src/python/vcf_snpeff_annotation.py -s GA5 -i build/resources/integrationTest/output/GA5/freebayes/GA5.freebayes.variants.vcf -o build/resources/integrationTest/output/GA5/freebayes/GA5.freebayes.variants.pass.annotation.tsv -t freebayes --canonical /ngs/data/reference_genome/GRCh37/Annotation/prefer_ensembl_transcript.txt --transvar /usr/bin/transvar -r hg19 --snpsift /usr/bin/snpsift --snpsift_db /ngs/data/tools/SnpEff/snpEff_v4.3p/snpEff/db --dbnsfp --dbsnp --coding_only --pass_only
if [ $? -eq 0 ]
then
	echo `date` Successful Step: SnpEff annotation.
	sleep 8
else
	echo `date` Error Step: SnpEff annotation.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: Remove temporary directories...

if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Finish the job execution!
