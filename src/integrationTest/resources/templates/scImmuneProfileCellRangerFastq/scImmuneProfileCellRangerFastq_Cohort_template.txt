# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

logFile=build/resources/integrationTest/output/log_files/scImmuneProfile_CellRanger_Fastq_alignment_for_pbmc4k_analysis.log
str=""
while [[('[')]][ $str = "" ]]
do
if [[('[')]][ -f $logFile  ]];
then
	str=$(grep -E "((Error Step: (Cellranger vdj analysis|Remove temporary directories))|(Successful Step: Cellranger vdj analysis))" $logFile;)
fi
	echo Waiting for step: Cellranger vdj analysis
	sleep 60
done

if [[('[')]][ $str == "*Error Step: Cellranger vdj analysis*" ]];
then
	echo $(date) Error QC results from pbmc4k:
	echo $str
else
	echo $(date) Confirm QC results from pbmc4k
fi

echo `date` Begin Step: QC summary analysis...
/ngs/data/app/R/v3.5.0/bin/Rscript [(${jarPath})]/src/R/QC_summary_analysis.R -i build/resources/integrationTest/immuneFastqList.tsv -d build/resources/integrationTest/output -w scImmuneProfile_CellRanger_Fastq
if [ $? -eq 0 ]
then
	echo `date` Successful Step: QC summary analysis.
	sleep 8
else
	echo `date` Error Step: QC summary analysis.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Finish the job execution!
