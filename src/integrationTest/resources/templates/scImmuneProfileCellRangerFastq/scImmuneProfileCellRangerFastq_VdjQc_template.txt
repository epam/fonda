# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: Cellranger vdj analysis...
cd build/resources/integrationTest/output/vdj
/ngs/data/tools/cellranger/v2.1.1/cellranger vdj --localcores=8 --id=pbmc4k --reference=/ngs/data/tools/cellranger/references/GRCh38-alts-ensembl-2.0.0-vdj_2018_08_07 --fastqs=[(${fastqDir})] --sample=pbmc4k --force-cells=5000
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Cellranger vdj analysis.
	sleep 8
else
	echo `date` Error Step: Cellranger vdj analysis.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Begin Step: Remove temporary directories...

if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Finish the job execution!
