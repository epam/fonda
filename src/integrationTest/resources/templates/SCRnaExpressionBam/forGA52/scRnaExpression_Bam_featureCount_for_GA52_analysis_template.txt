cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: Featurecounts...
feature_count -F SAF -M -s 0 -a /ngs/data/reference_genome/GRCh38/Annotation/Gencode_v26/gencode.v26.annotation.saf -o build/resources/integrationTest/output/GA52/feature_count/GA52_featureCount_gene.txt -Q 20 -T 4 /ngs/data/demo/test/fastq_data/GA5_0001_L002_R1_002.bam
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Run featureCount.
	sleep 8
else
	echo `date` Error Step: Run featureCount.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: Remove temporary directories...
rm -rf build/resources/integrationTest/output/GA52/tmp

if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Finish the job execution!
