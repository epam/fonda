#$ -pe threaded 4
#$ -m a

# --- The commands to be executed --- #

cd build/resources/integrationTest/output

echo `date` Begin the job execution...

echo `date` Begin Step: Scalpel detection...
/usr/bin/scalpel/scalpel-discovery --single --ref /ngs/data/reference_genome/hg19/hg19_decoy/hg19.decoy.fa --bam build/resources/integrationTest/output/GA5/bam/GA5.merged.sorted.bam --bed /ngs/data/S03723314_Padded.bed --dir build/resources/integrationTest/output/GA5/scalpel --format vcf --mapscore 10 --numprocs 8 --covthr 5 --mincov 5 --outratio 0.05
mv build/resources/integrationTest/output/GA5/scalpel/variants.indel.vcf build/resources/integrationTest/output/GA5/scalpel/GA5.scalpel.variants.vcf
if [ $? -eq 0 ]
then
	echo `date` Successful Step: Scalpel detection.
	sleep 8
else
	echo `date` Error Step: Scalpel detection.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: SnpEff annotation...
/usr/bin/python [(${jarPath})]/src/python/vcf_snpeff_annotation.py -s GA5 -i build/resources/integrationTest/output/GA5/scalpel/GA5.scalpel.variants.vcf -o build/resources/integrationTest/output/GA5/scalpel/GA5.scalpel.variants.pass.annotation.tsv -t scalpel --canonical /ngs/data/reference_genome/GRCh37/Annotation/prefer_ensembl_transcript.txt --transvar /usr/bin/transvar -r hg19 --snpsift /usr/bin/snpsift --snpsift_db /ngs/data/tools/SnpEff/snpEff_v4.3p/snpEff/db --dbnsfp --dbsnp --coding_only --pass_only
if [ $? -eq 0 ]
then
	echo `date` Successful Step: SnpEff annotation.
	sleep 8
else
	echo `date` Error Step: SnpEff annotation.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi

echo `date` Begin Step: Remove temporary directories...

if [ $? -eq 0 ]
then
	echo `date` Successful Step: Remove temporary directories.
	sleep 8
else
	echo `date` Error Step: Remove temporary directories.

	echo `date` The job was aborted due to ERRORS found.
	exit 1;
fi
echo `date` Finish the job execution!
